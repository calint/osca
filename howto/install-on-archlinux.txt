# https://wiki.archlinux.org/title/Installation_guide
# after "arch-chroot /mnt" and before booting into the installation

# install text editors
pacman -S vi nano

# copy network config
cp /etc/systemd/network/* /mnt/etc/systemd/network/ 

# after booting

# bring eth up
ip link set enp0s3 up

# enable and start networking services
systemctl enable systemd-networkd
systemctl start systemd-networkd
systemctl status systemd-networkd
systemctl enable systemd-resolved
systemctl start systemd-resolved
systemctl status systemd-resolved

#  then

# create user 'c' and home directory
mkdir /home/c
useradd -d /home/c c
chown -R c:c /home/c
passwd c

# enable 'sudo'
pacman -S sudo

nano /etc/sudoers

# uncomment:
# %wheel ALL=(ALL) NOPASSWD: ALL 

gpasswd -a c wheel

# logout root, login user

mkdir w
cd w
pacman -S git
git clone https://github.com/calint/a
cd a/osca
cp -rav etc/skel/. ~/

# log-out log-in

# compiler
sudo pacman -S gcc

# X11
sudo pacman -S xorg xorg-xinit xterm

# clonky
sudo pacman -S libxft acpi sysstat

# frameless launched applications
sudo pacman -S ttf-dejavu chromium thunar leafpad

# sticky
sudo pacman -S pkg-config gtk3 gtk2

cd /home/c/w/a/osca
sh make-all.sh

sudo pacman -S wget pcmanfm feh scrot mplayer pkgfile alsa-utils
